country_event = {
	id = ups.1
	title = ups.1.t
	desc = ups.1.d
	picture = GFX_report_event_japan_army_mountainside
	
	is_triggered_only = yes
	
	option = {
		name = ups.1.a
		ai_chance = { factor = 80 }
		add_ideas = UPS_soutient_partie_kyanhg
		
		#guerre civil fascism

		hidden_effect = {
			set_capital = { state = 195 }
			start_civil_war = {
				ruling_party = kyanhg
				ideology = fascism
				size = 0.2
				capital = 206
				states = { 204 194 241 }
			}
			random_other_country = {
				limit = {
					original_tag = UPS
					has_government = fascism
				}
				set_stability = 0.3
				set_war_support = 0.8
			}
			hidden_effect = { news_event = { id = news.grandleader.5 hours = 1 } }
			
			add_popularity = {
				ideology = kyanhg
				popularity = 0.60
			}

			set_stability = 0.3
			set_war_support = 0.8
		}
		custom_effect_tooltip = UPS_fascism_civil_war_tt
		add_stability = -0.25
		random_other_country = {
			limit = {
				original_tag = UPS
				has_government = fascism
			}
			set_cosmetic_tag = GER_civil_war
		}
		hidden_effect = {
			random_other_country = {
				limit = {
					original_tag = UPS
					has_government = fascism
				}
				save_event_target_as = fascist_ups_civil_war
				every_unit_leader = {
					set_nationality = UPS
				}
			}
			UPS = {
				every_unit_leader = {
					limit = {
						OR = {
							has_id = 3010
						}
					}
					set_nationality = event_target:fascist_ups_civil_war
				}
			}
		}		
	}

	option = {
		name = ups.1.b
		ai_chance = { factor = 20 }
		stability_factor = -0.1
		add_ideas = UPS_armee_faible
		add_ideas = UPS_troubles_peuple
		remove_ideas = UPS_soutient_partie_kyanhg
	}
}
